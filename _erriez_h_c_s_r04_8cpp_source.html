78 int16_t ErriezHCSR04::getDistance()
79 {
80  uint16_t distance;
81 
82  // Wait at least 2 microseconds with TRIG pin low before starting measurement
83  HCSR04_TRIG_LOW();
84  delayMicroseconds(2);
85 
86  // Generate 10 ms pulse TRIG pin
87  HCSR04_TRIG_HIGH();
88  delayMicroseconds(10);
89  HCSR04_TRIG_LOW();
90 
91  // Reads the ECHO pin, returns the sound wave travel time in microseconds
92  // Distance = (high level time * velocity (340M/S)) / 2;
93  distance = (pulseIn(_echoPin, HIGH) * 0.034) / 2;
94 
95  // Max range is around 400cm
96  if (distance > 400) {
97  // Invalid distance
98  distance = -1;
99  }
100 
101  return distance;
102 }
